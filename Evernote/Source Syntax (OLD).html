program: list(programDecl)


programDecl:
  • Feature(featureDecl)
  • Type(typeDecl)
  • Data(dataDecl)


dataDecl:
  1. Name: upperIdenitifier
  2. Value: dataExpr                       %No variables, functions, anonymous, or bindings


typeDecl:
  • TypeClass(classDecl)
  • TypeInterface(interfaceDecl)
  • TypeAlias(lowerIdentifier NEW, IowerIdentifier OLD)


featureDecl:
  1. Name: upperIdentifier
  2. FeatureElements: list(featureElement)


interfaceDecl:
  1. Name: Identifier
  2. GenericArgs: list(typeDef) % Could be empty list
  3. ClassList: list(headerDecl) % Should atleast have 1, Generic declaration inside class should be within these types


classDecl:
  1. Header: headerDecl
  2. Contractor: body
  3. Elements: plist(functionDecl)  % Should be atleast 1


channel:
  1. Type: typeDef % No Generics
  2. Name: Identifier % Identifier is upper


featureElement:
  • predicateDecl(predicateDecl)
  • functionDecl(functionDecl)
  • procedureDecl(maybe(header), body)
  • inputChannel(maybe(typeDef), upperIdentifier)


predicateDecl:
  1. Header: header
  2. Body: body


functionDecl:
  1. Header: header
  2. Body: body
  3. ReturnExpr: expr
  4. ReturnType: maybe(typeDef)


header:
  1. Name: upperIdentifier
  2. Arguments: list(argParam)


headerDecl:
  1. Name: upperIdentifier
  2. GenericArgs: list(upperIdentifier) % Could be empty list
  3. Arguments: list(typeParam)


typeParam:
  1. Type: typeDef
  2. Name: maybe(identifier)


argParam:
  1. Type: maybe(typeDef)
  2. Name: identifier


body: list(list(statement))


statement:
  • if(expr, block, maybe(block))
  • switch(expr, list(caseblock), maybe(caseblock) Default) 
  • stop
  • throw(expr)
  • patternMatch(expr, expr)
  • FileGenerate(expr NAME, expr CONTENTS)
  • GetData(identifier, expr) % Call or channel
  • PutData(indentifier, expr)
  • Error(expr) % Call another channel variable
  • log(expr)
  • continue(expr)
  • break(expr)
  • member(expr, expr)
  • predicateDecl(predicateDecl)
  • functionDecl(functionDecl)
  • for(identifier, expr, list(loopCase), block)
  • loop(identifier, expr, block)
  • statement(expr) % Should be a func call to evaluate boolean or a predicate.


expr: pureExpr, pattern, predicateArg, Assignment
  • binary(op2, expr, expr)
  • unary(op1, expo)
  • Primitive(primitive)
  • call(upperIdentifier, list(expr)) %only func call
  • upperIdentifier(UpperIdenitifier)
  • if(expr CONDITION, block THEN, maybe(block) ELSE)
  • term(maybe(lowerIdentifier), list(expr))
  • anonymous(identifier)
  • singleBind(Identifier) : $
  • multiBug(Identifier): ?
  • for(identifier, expr, list(loopCase), block)
  • loop(identifier, expr, block)
  • unique(expire, block)
  • seq(expo, block)


caseBlock:
  1. Pattern:expr
  2. Condition: maybe(expr)
  3. Block: block


loopCase:
  1. Identifier
  2. expr


typeDef:
  1. Name: basicTypeDef
  2. Args: maybe(list(typeDef))


basicTypeDef:
  1. Name: identifier
  2. GenericArgs: maybe(list(basicTypeDef))


op1: unary, negative

op2:
  • Add
  • Sub
  • Mod
  • Mult
  • Div
  • Neq
  • Gt
  • Ge
  • Le
  • Lt
  • concat
  • joinlist
  • range
  • In
  • <<
  • >>


dataExpr: 
  • DataTerm(dataExpr)
  • Primitive(primitive)


dataTerm(X): term(maybe(lowerIdentifier), list(X))


primitive:
  • Char
  • Int
  • Bool
